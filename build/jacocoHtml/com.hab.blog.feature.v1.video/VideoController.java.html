package com.hab.blog.feature.v1.video;

import com.hab.blog.feature.v1.entities.video.Person;
import com.hab.blog.feature.v1.entities.video.VideoType;
import com.hab.blog.feature.v1.video.DTO.*;
import io.swagger.v3.oas.annotations.Operation;
import io.swagger.v3.oas.annotations.Parameter;
import io.swagger.v3.oas.annotations.tags.Tag;
import lombok.RequiredArgsConstructor;
import org.springframework.data.domain.PageRequest;
import org.springframework.data.domain.Pageable;
import org.springframework.data.domain.Sort;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.*;

import java.util.List;

@RestController
@RequestMapping("/api/v1/vision")
@Tag(name = "Video API", description = "视频内容管理接口")
@RequiredArgsConstructor
public class VideoController {

//    private final VideoService videoService;

    private final VideoStaticService videoStaticService;

    @GetMapping("/video/static")
    @Operation(summary = "获取预设视频数据", description = "适用于Vision Pro的本地静态视频预览")
    public ResponseEntity<List<VideoDTO>> getStaticVideos() {
        return ResponseEntity.ok(videoStaticService.getStaticVideos());
    }

    @GetMapping("/genre/static")
    @Operation(summary = "获取预设视频数据", description = "适用于Vision Pro的本地静态视频预览")
    public ResponseEntity<List<GenreDTO>> getStaticGenre() {
        return ResponseEntity.ok(videoStaticService.getAllGenres());
    }
    @GetMapping("/person/static")
    @Operation(summary = "获取预设视频数据", description = "适用于Vision Pro的本地静态视频预览")
    public ResponseEntity<List<PersonDTO>> getStaticPerson() {
        return ResponseEntity.ok(videoStaticService.getAllPeople());
    }

    @GetMapping("/relationship/static")
    @Operation(summary = "获取预设视频数据", description = "适用于Vision Pro的本地静态视频预览")
    public ResponseEntity<List<VideoRelationshipDTO>> getStaticRelationship() {
        return ResponseEntity.ok(videoStaticService.getAllRelationships());
    }

    @Operation(summary = "获取 Up Next 项列表", description = "提供本地静态 up-next 视频预览")
    @GetMapping("/upNext/static")
    public ResponseEntity<List<UpNextItemDTO>> getUpNextItems() {
        return ResponseEntity.ok(videoStaticService.getUpNextItems());
    }

    @GetMapping("/category/static")
    @Operation(summary = "获取视频分类", description = "返回静态定义的所有视频分类列表")
    public ResponseEntity<List<CategoryDTO>> getStaticCategories() {
        return ResponseEntity.ok(videoStaticService.getAllCategories());
    }
}
