package com.hab.blog.feature.v1.entities.video;

import jakarta.persistence.*;
import lombok.Data;

import java.util.HashSet;
import java.util.Set;

@Entity
@Table(name = "video")
@Data
public class Video {
    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    private String name;
    private VideoType videoType;
    private String url;
    private String imageName;

    @Column(columnDefinition = "TEXT")
    private String synopsis;

    @ManyToMany
    @JoinTable(
            name = "video_genre",
            joinColumns = @JoinColumn(name = "video_id"),
            inverseJoinColumns = @JoinColumn(name = "genre_id")
    )
    private Set<Genre> genres = new HashSet<>();

    @ManyToMany
    @JoinTable(
            name = "video_actor",
            joinColumns = @JoinColumn(name = "video_id"),
            inverseJoinColumns = @JoinColumn(name = "person_id")
    )
    private Set<Person> actors = new HashSet<>();
    @ManyToMany
    @JoinTable(
            name = "video_director",
            joinColumns = @JoinColumn(name = "video_id"),
            inverseJoinColumns = @JoinColumn(name = "person_id")
    )
    private Set<Person> directors = new HashSet<>();

    @ManyToMany
    @JoinTable(
            name = "video_writer",
            joinColumns = @JoinColumn(name = "video_id"),
            inverseJoinColumns = @JoinColumn(name = "person_id")
    )
    private Set<Person> writers = new HashSet<>();

    @ManyToOne(fetch = FetchType.LAZY)
    @MapsId("categoryId")
    @JoinColumn(name = "category_id")
    private Category category;

    private Integer yearOfRelease;
    private Integer duration;
    private Long startTime;

    @Enumerated(EnumType.STRING)
    private ContentRating contentRating;

    private Boolean isHero;
    private Boolean isFeatured;
    private Boolean isLocal;

    // 技术元数据
    private String resolution;
    private String videoCodec;
    private String audioCodec;
    private Long fileSize;

    // Vision Pro专用字段
    @Enumerated(EnumType.STRING)
    private SpatialType spatialType;

    @Enumerated(EnumType.STRING)
    private ProjectionType projectionType;

    @Enumerated(EnumType.STRING)
    private DepthFormat depthFormat;

    private String coordinateSystem;

    private Float fovVertical;  // 单位：度
    private Float fovHorizontal;

    // 空间原点坐标
    private Float spatialAnchorX;
    private Float spatialAnchorY;
    private Float spatialAnchorZ;
}
