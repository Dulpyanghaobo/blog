package com.hab.blog.feature.v1.space;

import com.hab.blog.feature.v1.entities.space.SpaceEntity;
import io.swagger.v3.oas.annotations.tags.Tag;
import lombok.RequiredArgsConstructor;
import org.springframework.web.bind.annotation.*;

@RestController
@RequestMapping("/api/v1/admin/spaces")
@RequiredArgsConstructor
@Tag(name = "Space Admin")
public class SpaceAdminController {

    private final SpaceService spaceService;

    @PostMapping
    public SpaceEntity create(@RequestBody SpaceEntity e) { return spaceService.create(e); }

    @PutMapping("/{id}")
    public SpaceEntity update(@PathVariable String id, @RequestBody SpaceEntity e) {
        return spaceService.update(id, e);
    }

    @DeleteMapping("/{id}")
    public void delete(@PathVariable String id) { spaceService.delete(id); }
}
