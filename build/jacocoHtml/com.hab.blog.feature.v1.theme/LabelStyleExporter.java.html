    public static void exportLabelStylesAsJsonFiles(Map<String, Object> allStyles, String outputFolder) {
        ObjectMapper mapper = new ObjectMapper().enable(SerializationFeature.INDENT_OUTPUT);
        new File(outputFolder).mkdirs();  // ensure directory exists

        for (Map.Entry<String, Object> entry : allStyles.entrySet()) {
            String styleName = entry.getKey();
            Object value = entry.getValue();
            File outFile = new File(outputFolder, styleName + ".json");
            try {
                mapper.writeValue(outFile, value);
                System.out.println("✅ 已生成: " + outFile.getPath());
            } catch (IOException e) {
                System.err.println("❌ 写入失败: " + styleName);
            }
        }
    }
